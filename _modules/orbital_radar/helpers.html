    """
    Conversion from dB to linear.

    Parameters

    """

[docs]def li2db(x):
    """
    Conversion from linear to dB.

    Parameters
    ----------
    x : float
        Any value or array to be converted from linear to dB unit
    """
    return 10 * np.log10(x)
